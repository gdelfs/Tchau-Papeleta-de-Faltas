CREATE SCHEMA tchau_papeleta;
CREATE TABLE tchau_papeleta.turma(nome_turma varchar(10) NOT NULL ,ano varchar(3), secao_ensino varchar (5), PRIMARY KEY(nome_turma));
CREATE TABLE tchau_papeleta.aluno(matricula varchar(5) NOT NULL, nome varchar (30), posto varchar (5), nome_turma varchar(10), PRIMARY KEY(matricula), FOREIGN KEY(nome_turma) REFERENCES tchau_papeleta.turma(nome_turma));
CREATE TABLE tchau_papeleta.professor(cod_professor varchar(5) NOT NULL, nome varchar (30) , posto varchar (5), PRIMARY KEY(cod_professor) );
CREATE TABLE tchau_papeleta.disciplina(cod_disciplina varchar(10) NOT NULL , materia varchar(45), nome_turma varchar(10) NOT NULL, cod_professor1 varchar(5) NOT NULL, cod_professor2 varchar(5), PRIMARY KEY(cod_disciplina), FOREIGN KEY(nome_turma) REFERENCES tchau_papeleta.turma(nome_turma), FOREIGN KEY(cod_professor1) REFERENCES tchau_papeleta.professor(cod_professor) , FOREIGN KEY(cod_professor2) REFERENCES tchau_papeleta.professor(cod_professor));
CREATE TABLE tchau_papeleta.aula(cod_aula varchar(20) NOT NULL, dia integer, mes integer, ano integer, cod_disciplina varchar(10) NOT NULL, tempo_aula integer, nome_foto varchar (100), primary key(cod_aula), FOREIGN KEY(cod_disciplina) REFERENCES tchau_papeleta.disciplina(cod_disciplina));
CREATE TABLE tchau_papeleta.papeleta(cod_papeleta varchar(23) NOT NULL, cod_aula varchar(20) NOT NULL, matricula varchar(5) NOT NULL, presente boolean, PRIMARY KEY(cod_papeleta), FOREIGN KEY(cod_aula) REFERENCES tchau_papeleta.aula(cod_aula), FOREIGN KEY(matricula) REFERENCES tchau_papeleta.aluno(matricula));
